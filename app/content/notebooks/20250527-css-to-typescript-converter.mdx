# OKLCH to RGB hex approximation
```python
def oklch_to_rgb_hex(L, C, h):
 h_rad = math.radians(h)
 a = math.cos(h_rad) * C
 b = math.sin(h_rad) * C
```
    # OKLab intermediate
```python
 l_ = L
 m = l_ + 0.3963377774 * a + 0.2158037573 * b
 s = l_ - 0.1055613458 * a - 0.0638541728 * b
 l = l_ - 0.0894841775 * a - 1.2914855480 * b
```
    # LMS to linear sRGB
```python
 r = 4.0767416621 * l - 3.3077115913 * m + 0.2309699292 * s
 g = -1.2684380046 * l + 2.6097574011 * m - 0.3413193965 * s
 b = -0.0041960863 * l - 0.7034186147 * m + 1.7076147010 * s
```
    # Clamp and gamma correct
```python
 r = max(0, min(1, r)) ** (1/2.2)
 g = max(0, min(1, g)) ** (1/2.2)
 b = max(0, min(1, b)) ** (1/2.2)
 return mcolors.to_hex([r, g, b])
```

# Function to determine ideal text color for contrast (black or white)
```python
def get_contrast_text_color(hex_color):
 r, g, b = mcolors.to_rgb(hex_color)
 luminance = 0.299 * r + 0.587 * g + 0.114 * b
```
